# Источники 

```dataviewjs
const pages = dv.pages('"Источники"')
  .where(p => p.file.ext === "md" && (() => {
    const персоны = p["Персоны"];
    if (!персоны) return false;
    return Array.isArray(персоны) 
      ? персоны.some(link => link.path === dv.current().file.path)
      : персоны.path === dv.current().file.path;
  })())
  .sort(p => p["Год"], 'asc');

function formatDate(dateValue) {
  if (!dateValue) return "";
  
  // Преобразуем в объект Date
  const date = new Date(dateValue);
  
  if (isNaN(date)) return ""; // если невалидная дата
  
  // dd.MM.yyyy
  const day = String(date.getDate()).padStart(2, '0');
  const month = String(date.getMonth() + 1).padStart(2, '0');
  const year = date.getFullYear();
  
  return `${day}.${month}.${year}`;
}

for (let page of pages) {
  const год = page["Год"] || '';
  const датаСыр = page["Дата события"] || '';
  // Форматируем дату события если она есть
  const датаСобытия = датаСыр ? formatDate(датаСыр) : '';
  const заголовок = датаСобытия || год;
  
  dv.header(3, `${заголовок} - ` + page.file.link);
  
  const content = await dv.io.load(page.file.path);
  const cleanContent = content.replace(/^---[\s\S]*?---\n?/i, '').trim();
  
  dv.paragraph(cleanContent);
  dv.paragraph("---");
}
```
